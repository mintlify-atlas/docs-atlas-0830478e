## Overview

The `TransactionService` provides methods for creating, reading, updating, and deleting transactions, as well as generating monthly reports. All methods return RxJS Observables.

## Properties

<ParamField path="apiUrl" type="string" default="${env.myFinanceApiUrl}/Transaction">
  Base URL for transaction API endpoints
</ParamField>

## Methods

### createTransaction()

Creates a new transaction.

```typescript
createTransaction(transaction: createTransactionModel): Observable<ApiResponse<Transaction>>
```

<ParamField path="transaction" type="createTransactionModel" required>
  Transaction data to create
</ParamField>

<ResponseField name="returns" type="Observable<ApiResponse<Transaction>>">
  Observable emitting the created transaction wrapped in ApiResponse
</ResponseField>

**Example:**

```typescript
const newTransaction: createTransactionModel = {
  amount: 150.00,
  type: 'Expense',
  description: 'Grocery shopping',
  categoryId: 5,
  date: '2026-03-05'
};

transactionService.createTransaction(newTransaction).subscribe({
  next: (response) => console.log('Created:', response.data),
  error: (err) => console.error('Error:', err)
});
```

### getMonthlyReport()

Generates a monthly report for income or expenses for a given year.

```typescript
getMonthlyReport(type: 'Income' | 'Expense', year: number): Observable<ApiResponse<Report[]>>
```

<ParamField path="type" type="'Income' | 'Expense'" required>
  Type of transactions to report on
</ParamField>

<ParamField path="year" type="number" required>
  Year for the report (e.g., 2026)
</ParamField>

<ResponseField name="returns" type="Observable<ApiResponse<Report[]>>">
  Observable emitting array of monthly report data
</ResponseField>

**Example:**

```typescript
transactionService.getMonthlyReport('Expense', 2026).subscribe({
  next: (response) => {
    response.data.forEach(report => {
      console.log(`${report.month}/${report.year}: $${report.total}`);
    });
  }
});
```

## Source Code

Location: `src/app/services/transaction-service.ts`

```typescript
import { HttpClient } from "@angular/common/http";
import { Injectable } from "@angular/core";
import { env } from "../../environments/environment";
import { createTransactionModel, Transaction } from "../models/transaction-model";
import { Observable } from "rxjs/internal/Observable";
import { ApiResponse } from "../models/apiResponse.model";
import { Report } from "../models/report-model";

@Injectable({
  providedIn: 'root'
})
export class TransactionService {
    private apiUrl = `${env.myFinanceApiUrl}/Transaction`;
    constructor(private http: HttpClient) { }

    createTransaction(transaction: createTransactionModel): Observable<ApiResponse<Transaction>> {
        return this.http.post<ApiResponse<Transaction>>(this.apiUrl, transaction);
    }
    
    getAllTransactions(): Observable<ApiResponse<Transaction[]>> {
        return this.http.get<ApiResponse<Transaction[]>>(this.apiUrl);
    }
    
    getTransactionById(id: string): Observable<ApiResponse<Transaction>> {
        return this.http.get<ApiResponse<Transaction>>(`${this.apiUrl}/${id}`);
    }
    
    updateTransaction(id: string, transaction: Transaction): Observable<ApiResponse<Transaction>> {
        return this.http.put<ApiResponse<Transaction>>(`${this.apiUrl}/${id}`, transaction);
    }
    
    deleteTransaction(id: string): Observable<void> {
        return this.http.delete<void>(`${this.apiUrl}/${id}`);
    }
    
    getMonthlyReport(type: 'Income' | 'Expense', year: number): Observable<ApiResponse<Report[]>> {
        return this.http.get<ApiResponse<Report[]>>(`${this.apiUrl}/report?type=${type}&year=${year}`);
    }
}
```
